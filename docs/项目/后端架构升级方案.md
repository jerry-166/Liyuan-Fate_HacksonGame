> **版本**: v2.0 | **日期**: 2026-05-25
> **基于**: 之前架构讨论的决策 + 《梨园生死》完整剧情文档（六章结构）

## 二、新数据模型

### 2.1 SubTask（子任务）

```python
from enum import Enum
from dataclasses import dataclass, field
from typing import Optional

class SubTaskMode(str, Enum):
    DIALOGUE = "dialogue"            # 与某个 NPC 对话达到深度
    ACQUIRE_ITEM = "acquire_item"    # 获取某件物品
    SHOW_ITEM = "show_item"          # 向某个 NPC 展示某件物品
    DELIVER = "deliver"              # 从 NPC A → 主角 → NPC B 的传递链
    RELATION_THRESHOLD = "relation"  # 与某个 NPC 达到特定关系值
    EXPLORE = "explore"              # 到达某个地点/场景

class SubTaskStatus(str, Enum):
    LOCKED = "locked"
    ACTIVE = "active"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"

@dataclass
class SubTask:
    id: str
    title: str
    mode: SubTaskMode
    description: str = ""
    # 根据 mode 动态填充
    target_npc_id: Optional[str] = None        # DIALOGUE / SHOW_ITEM / DELIVER 来源
    deliver_to_npc_id: Optional[str] = None    # DELIVER 目标
    required_item_id: Optional[str] = None     # ACQUIRE_ITEM / SHOW_ITEM / DELIVER
    target_scene: Optional[str] = None         # EXPLORE
    relation_threshold: Optional[int] = None   # RELATION_THRESHOLD
    # 状态
    status: SubTaskStatus = SubTaskStatus.LOCKED
```

### 2.2 TaskInstance（任务实例）

```python
@dataclass
class TaskInstance:
    task_id: str
    chapter_id: str
    chapter_name: str
    description: str              # LLM 生成的任务描述
    sub_tasks: list[SubTask]
    related_npc_ids: list[str]    # 所有参与此章节的 NPC
    generated_item_ids: list[str] # LLM 可能动态生成的物品
    npc_completion_votes: dict[str, bool] = field(default_factory=dict)
    # NPC 投票：True=认为完成, False=认为未完成

    @property
    def is_completed(self) -> bool:
        if not self.related_npc_ids:
            return True
        return all(self.npc_completion_votes.get(npc_id, False)
                   for npc_id in self.related_npc_ids)

    @property
    def completion_rate(self) -> float:
        total = len(self.related_npc_ids)
        if total == 0: return 1.0
        return sum(1 for v in self.npc_completion_votes.values() if v) / total
```

### 2.3 NarrativeItem（叙事物品）

```python
@dataclass
class NarrativeItem:
    id: str
    name: str
    item_type: str                # clothing / document / misc / instrument / container
    base_description: str         # 固定描述（来自 story_items.yaml）
    ai_detail: Optional[str] = None      # AI 生成的补充详情
    ai_detail_locked: bool = False       # flag: 一旦锁定不再变更
    is_key: bool = False                 # 是否为关键叙事道具
    location: Optional[dict] = None      # {scene, position}
    is_discovered: bool = False          # 玩家是否已发现
    discovery_context: str = ""          # 发现时的旁白文本
    related_npcs: list[str] = field(default_factory=list)  # 关联 NPC
    npc_knowledge: dict[str, str] = field(default_factory=dict)  # {npc_id: 该NPC对此物品的认知}
    desc_source: str = "fixed"     # "fixed" | "ai"
    stage_relevance: list[int] = field(default_factory=list)
    source_npc: Optional[str] = None  # 由哪个 NPC 给予
```

## 三、数据库 Schema 变更

### 3.1 新增表

```sql
-- 叙事物品表
CREATE TABLE IF NOT EXISTS narrative_items (
    id TEXT PRIMARY KEY,
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    item_type TEXT NOT NULL,
    name TEXT NOT NULL,
    base_description TEXT DEFAULT '',
    ai_detail TEXT,
    ai_detail_locked INTEGER DEFAULT 0,
    is_key INTEGER DEFAULT 0,
    is_discovered INTEGER DEFAULT 0,
    discovery_context TEXT DEFAULT '',
    related_npcs TEXT DEFAULT '[]',       -- JSON array
    npc_knowledge TEXT DEFAULT '{}',       -- JSON object {npc_id: text}
    desc_source TEXT DEFAULT 'fixed',
    location_scene TEXT,
    location_pos TEXT DEFAULT '{}',        -- JSON {x, y}
    source_npc TEXT,
    stage_relevance TEXT DEFAULT '[]',     -- JSON array
    UNIQUE(session_id, id)
);

-- 任务实例表
CREATE TABLE IF NOT EXISTS task_instances (
    id TEXT PRIMARY KEY,
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    chapter_id TEXT NOT NULL,
    chapter_name TEXT,
    description TEXT,
    sub_tasks TEXT DEFAULT '[]',           -- JSON SubTask[]
    related_npc_ids TEXT DEFAULT '[]',
    npc_completion_votes TEXT DEFAULT '{}',-- JSON {npc_id: bool}
    is_completed INTEGER DEFAULT 0,
    created_at TEXT DEFAULT (datetime('now'))
);

-- 章节进度表
CREATE TABLE IF NOT EXISTS chapter_progress (
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    chapter_id TEXT NOT NULL,
    task_id TEXT REFERENCES task_instances(id),
    status TEXT DEFAULT 'active',          -- active / completed
    started_at TEXT DEFAULT (datetime('now')),
    completed_at TEXT,
    PRIMARY KEY(session_id, chapter_id)
);
```

| 1 | POST | `/api/game/{id}/chapter/start` | 开始下一章（触发 LLM 任务规划） | P0 |

## 八、风险与应对

| 风险 | 等级 | 应对 |
|------|------|------|
| **STAGES 迁移破坏现有对话** | 🔴 | 保留 STAGES 兼容层，`current_stage` 从 `current_chapter_id` 派生 |
| **NPC 数量增加 → Token 预算紧张** | 🟡 | 分层注入策略，对话时只加载相关 NPC+任务+物品 |
| **LLM 任务规划质量不稳定** | 🟡 | sub_task_templates 作为硬兜底，LLM 结果需要验证层校验 |
| **物品发现时序问题** | 🟡 | 物品按 chapter 的 stage_relevance 控制，未到章节不可发现 |
| **NPC 投票逻辑过于严格** | 🟢 | 支持权重投票：关键 NPC 票数更高，或仅核心 NPC 投票即可 |

## 九、与现有系统的兼容策略

```python
# 兼容层：从章节推导阶段（供前端色调/BGM等复用）
CHAPTER_TO_STAGE = {
    "ch_prologue": 1,
    "ch_01": 1,
    "ch_02": 1,
    "ch_03": 2,  # 真相觉醒 → 对应旧"了解"阶段
    "ch_04": 2,
    "ch_05": 3,  # 承戏重振 → 对应旧"抉择"阶段
}

# GameSession.current_stage 属性变为计算属性
@property
def current_stage(self) -> int:
    if self.current_chapter_id:
        return CHAPTER_TO_STAGE.get(self.current_chapter_id, 1)
    return 1
```

这样前端现有的色调切换、BGM 切换逻辑不需要大改，只需要额外读取 `chapter_name` 和 `color_tone` 即可。

### 10.4 剧本加载流程

```

│                                             │
```

### 10.5 ScriptLoader 核心实现

```python
# backend/data/script_loader.py
import yaml
from pathlib import Path
from typing import Optional

SCRIPTS_DIR = Path(__file__).parent.parent / "data" / "scripts"

class ScriptValidationError(Exception):
    """剧本数据完整性校验失败"""
    pass

class ScriptLoader:
    """剧本加载器 — 读取并校验一个剧本的全部数据"""
    
    def __init__(self, scripts_dir: Path = SCRIPTS_DIR):
        self.scripts_dir = scripts_dir
    
    def list_scripts(self) -> list[dict]:
        """列出所有可用剧本"""
        scripts = []
        for d in self.scripts_dir.iterdir():
            meta_file = d / "meta.yaml"
            if meta_file.exists():
                with open(meta_file, "r", encoding="utf-8") as f:
                    meta = yaml.safe_load(f)
                scripts.append({
                    "script_id": meta["script_id"],
                    "name": meta["name"],
                    "version": meta.get("version", "1.0"),
                    "npc_count": len(meta.get("npcs", [])),
                    "chapter_count": len(meta.get("chapters_file", "")),  # 实际加载后填充
                })
        return scripts
    
    def load_script(self, script_id: str) -> dict:
        """
        加载完整剧本数据，返回结构化字典。
        失败时抛出 ScriptValidationError。
        """
        script_dir = self.scripts_dir / script_id
        if not script_dir.exists():
            raise ScriptValidationError(f"剧本目录不存在: {script_dir}")
        
        # 1. 加载 meta
        meta = self._load_yaml(script_dir / "meta.yaml")
        
        # 2. 加载世界观 Prompt
        system_prompt = ""
        prompt_file = script_dir / meta.get("system_prompt_file", "system_base.txt")
        if prompt_file.exists():
            system_prompt = prompt_file.read_text(encoding="utf-8")
        else:
            # 回退到全局默认
            global_prompt = Path(__file__).parent.parent / "prompts" / "system_base.txt"
            system_prompt = global_prompt.read_text(encoding="utf-8") if global_prompt.exists() else ""
        
        # 3. 加载章节
        chapters = self._load_yaml(script_dir / meta["chapters_file"])
        
        # 4. 加载物品
        items = []
        items_file = script_dir / meta.get("items_file", "items/story_items.yaml")
        if items_file.exists():
            items = self._load_yaml(items_file)
        
        # 5. 加载 NPC 人设
        personas = {}
        persona_dir = script_dir / meta.get("persona_dir", "personas")
        for npc_def in meta.get("npcs", []):
            persona_file = persona_dir / npc_def["persona_file"]
            if persona_file.exists():
                personas[npc_def["id"]] = self._load_yaml(persona_file)
        
        # 6. 加载 Prompt 覆盖
        prompt_overrides = {}
        for key, file_path in meta.get("prompt_overrides", {}).items():
            if file_path:
                p = script_dir / file_path
                if p.exists():
                    prompt_overrides[key] = p.read_text(encoding="utf-8")
        
        # 7. 完整性校验
        self._validate(meta, chapters, items, personas)
        
        return {
            "meta": meta,
            "system_prompt": system_prompt,
            "chapters": chapters,
            "items": items,
            "personas": personas,
            "prompt_overrides": prompt_overrides,
        }
    
    def _validate(self, meta, chapters, items, personas):
        """校验剧本数据完整性"""
        npc_ids = {npc["id"] for npc in meta.get("npcs", [])}
        item_ids = {item["id"] for item in items}
        errors = []
        
        for ch in chapters:
            # 校验章节引用的 NPC
            for npc_id in ch.get("required_npcs", []):
                if npc_id not in npc_ids:
                    errors.append(f"章节 {ch['id']} 引用了不存在的 NPC: {npc_id}")
            
            # 校验章节引用的物品
            for item_id in ch.get("required_items", []):
                if item_id not in item_ids:
                    errors.append(f"章节 {ch['id']} 引用了不存在的物品: {item_id}")
            
            # 校验子任务模板中的 NPC/物品引用
            for st in ch.get("sub_task_templates", []):
                if st.get("target_npc_id") and st["target_npc_id"] not in npc_ids:
                    errors.append(f"章节 {ch['id']} 子任务 {st['id']} 引用无效 NPC: {st['target_npc_id']}")
                if st.get("required_item_id") and st["required_item_id"] not in item_ids:
                    errors.append(f"章节 {ch['id']} 子任务 {st['id']} 引用无效物品: {st['required_item_id']}")
        
        if errors:
            raise ScriptValidationError(
                f"剧本数据校验失败 ({len(errors)} 个错误):\n" + "\n".join(errors)
            )
    
    def _load_yaml(self, path: Path):
        with open(path, "r", encoding="utf-8") as f:
            return yaml.safe_load(f) or []
```

### 10.8 新建剧本的流程（内容创作者）

内容创作者（如人 C）想创建一个新剧本时，只需：

```
1. 在 data/scripts/ 下创建新目录
   data/scripts/my_new_story/

2. 编写 meta.yaml（剧本元信息 + NPC 定义）
3. 编写 chapters.yaml（章节定义）
4. 编写 items/story_items.yaml（物品定义）
5. 编写 personas/*.yaml（角色卡）

   ⚠️ ScriptLoader 会自动校验引用完整性
   如果有 NPC/物品引用缺失，会在启动时报错并列出具体问题
```
