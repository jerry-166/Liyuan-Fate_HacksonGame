# 《梨园生死》数据库设计

> 以 `backend/storage/schema.sql` 为唯一真相源。共 **11 张表**（v3 新增 saves）。
> **WAL 模式**：初始化时执行 `PRAGMA journal_mode=WAL;`。
> **外键约束**：`PRAGMA foreign_keys=ON;`。

---

## 建表语句

```sql
-- 启用 WAL 模式（应用启动时执行一次）
PRAGMA journal_mode=WAL;

-- sessions 表：会话元数据
CREATE TABLE IF NOT EXISTS sessions (
    session_id   TEXT PRIMARY KEY,
    player_name  TEXT NOT NULL DEFAULT '玩家',
    current_stage INTEGER NOT NULL DEFAULT 1,
    current_chapter_id TEXT,                -- v2: 当前章节ID
    script_id TEXT DEFAULT 'liyuan_shengsi', -- v2: 剧本ID（多剧本支持）
    active_item TEXT,                       -- v2: 当前选中物品
    game_ended   INTEGER NOT NULL DEFAULT 0,  -- 0/1
    ending_type  TEXT,
    ending_data  TEXT,                         -- JSON: 缓存的结局评价
    deleted      INTEGER NOT NULL DEFAULT 0,   -- 0/1: 软删除标记
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- dialogues 表：对话历史
CREATE TABLE IF NOT EXISTS dialogues (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    npc_id      TEXT NOT NULL,
    role        TEXT NOT NULL CHECK(role IN ('player', 'npc')),
    content     TEXT NOT NULL,
    options     TEXT,                           -- JSON: NPC 回复附带的对话选项
    stage       INTEGER,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE INDEX IF NOT EXISTS idx_dialogues_session
    ON dialogues(session_id, npc_id);

-- events 表：已触发事件日志
CREATE TABLE IF NOT EXISTS events (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    event_id    TEXT NOT NULL,
    description TEXT,
    triggered_by_npc TEXT,
    stage       INTEGER,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE INDEX IF NOT EXISTS idx_events_session
    ON events(session_id);

-- npc_states 表：NPC 运行时状态
CREATE TABLE IF NOT EXISTS npc_states (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    npc_id      TEXT NOT NULL,
    relationship     INTEGER NOT NULL DEFAULT 0,
    is_available     INTEGER NOT NULL DEFAULT 1,   -- 0/1
    current_greeting TEXT NOT NULL DEFAULT '',
    dialogue_round_count INTEGER NOT NULL DEFAULT 0,
    position_col INTEGER DEFAULT 0,               -- v3: 瓦片坐标列号
    position_row INTEGER DEFAULT 0,               -- v3: 瓦片坐标行号
    scene       TEXT NOT NULL DEFAULT '',          -- v3: 所在场景
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id),
    UNIQUE(session_id, npc_id)
);

CREATE INDEX IF NOT EXISTS idx_npc_states_session
    ON npc_states(session_id);

-- relationship_log 表：关系值变化日志
CREATE TABLE IF NOT EXISTS relationship_log (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    npc_id      TEXT NOT NULL,
    delta       INTEGER NOT NULL,          -- 本轮变化量（如 +5）
    old_value   INTEGER NOT NULL,          -- 变化前值
    new_value   INTEGER NOT NULL,          -- 变化后值
    reason      TEXT DEFAULT '',           -- 变化原因
    dialogue_id INTEGER,                   -- 关联的对话记录ID（可选）
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE INDEX IF NOT EXISTS idx_rel_log_session
    ON relationship_log(session_id, npc_id);

-- player_choices 表：玩家关键选择记录
CREATE TABLE IF NOT EXISTS player_choices (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    npc_id      TEXT NOT NULL,
    choice_text     TEXT NOT NULL,         -- 玩家选择的选项文本
    available_options TEXT,                 -- JSON: 当时所有可选选项
    dialogue_id INTEGER,                   -- 关联的对话记录ID
    stage       INTEGER,                   -- 选择时所在阶段
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE INDEX IF NOT EXISTS idx_choices_session
    ON player_choices(session_id);

-- stage_history 表：阶段切换历史
CREATE TABLE IF NOT EXISTS stage_history (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id  TEXT NOT NULL,
    from_stage  INTEGER NOT NULL,
    to_stage    INTEGER NOT NULL,
    reason      TEXT DEFAULT '',           -- "rule: xxx" 或 "llm: 连续N轮推进"
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

-- ═══ v2 新增表 ═══

-- narrative_items 表：叙事物品运行时实例
CREATE TABLE IF NOT EXISTS narrative_items (
    id TEXT PRIMARY KEY,
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    item_type TEXT NOT NULL DEFAULT 'misc',
    name TEXT NOT NULL,
    base_description TEXT DEFAULT '',
    ai_detail TEXT,                        -- LLM 生成的详细描述
    ai_detail_locked INTEGER DEFAULT 0,    -- 描述是否锁定
    is_key INTEGER DEFAULT 0,              -- 关键物品标记
    is_discovered INTEGER DEFAULT 0,       -- 是否已发现
    discovery_context TEXT DEFAULT '',     -- 发现旁白
    related_npcs TEXT DEFAULT '[]',        -- JSON: 关联NPC
    npc_knowledge TEXT DEFAULT '{}',       -- JSON: NPC对此物品的特殊认知
    desc_source TEXT DEFAULT 'fixed',
    location_scene TEXT,
    location_pos TEXT DEFAULT '{}',
    source_npc TEXT,
    stage_relevance TEXT DEFAULT '[]',     -- JSON: 相关阶段
    template_ref TEXT,
    holdable INTEGER DEFAULT 1,            -- 是否可持有
    acquire_method TEXT DEFAULT '',
    UNIQUE(session_id, id)
);

CREATE INDEX IF NOT EXISTS idx_narrative_items_session
    ON narrative_items(session_id);

-- task_instances 表：任务实例
CREATE TABLE IF NOT EXISTS task_instances (
    id TEXT PRIMARY KEY,
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    chapter_id TEXT NOT NULL,
    chapter_name TEXT,
    description TEXT,
    sub_tasks TEXT DEFAULT '[]',           -- JSON: 子任务列表
    related_npc_ids TEXT DEFAULT '[]',     -- JSON: 关联NPC
    npc_completion_votes TEXT DEFAULT '{}', -- JSON: NPC投票状态
    is_completed INTEGER DEFAULT 0,
    created_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_task_instances_session
    ON task_instances(session_id);

-- chapter_progress 表：章节进度
CREATE TABLE IF NOT EXISTS chapter_progress (
    session_id TEXT NOT NULL REFERENCES sessions(session_id),
    chapter_id TEXT NOT NULL,
    task_id TEXT REFERENCES task_instances(id),
    status TEXT DEFAULT 'active',          -- active / completed
    started_at TEXT DEFAULT (datetime('now')),
    completed_at TEXT,
    PRIMARY KEY(session_id, chapter_id)
);

-- ═══ v3 新增表 ═══

-- saves 表：存档快照元数据（1 Session : N Saves）
CREATE TABLE IF NOT EXISTS saves (
    save_id   TEXT PRIMARY KEY,            -- sv_{uuid}
    session_id TEXT NOT NULL,              -- 关联会话
    slot_id   INTEGER NOT NULL,            -- 槽位 1-6
    label     TEXT NOT NULL DEFAULT '',    -- 显示标签
    stage     INTEGER NOT NULL DEFAULT 1,  -- 存档时所在阶段
    chapter_id TEXT,                       -- 存档时所在章节
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id),
    UNIQUE(session_id, slot_id)            -- 同 session 下 slot_id 唯一
);
CREATE INDEX IF NOT EXISTS idx_saves_session ON saves(session_id);
```

> **v3 说明**: 存档的完整游戏状态以 JSON 快照形式存储在 `backend/storage/saves/{session_id}/{save_id}.json`，数据库仅存储轻量级元数据，便于列表查询。每个 session 最多 6 个槽位。
>
> **快照 JSON 包含的数据**（实现存档级数据隔离）：
> - **NPC 完整状态**：每个 NPC 的 `dialogue_history`（完整对话轮次，每条包含 `turn_index` 全局序号）、`last_options`、`relationship`、`is_available`、`current_greeting`、`dialogue_round_count`、`position`、`scene`
> - **全局 LLM 状态**：`stage_llm_consecutive`、`persona_cache`
> - **对话全局计数器**：`dialogue_turn_counter`（整型，自增，用于跨 NPC 时间排序）
> - **游戏进度**：`current_stage`、`current_chapter_id`、`completed_chapters`、`events_triggered`、`game_ended`、`ending_data`
> - **背包与任务**：`inventory`（完整 NarrativeItem）、`current_task`（TaskInstance + 子任务）
> - **位置快照**：主角 + 故事 NPC + 普通 NPC 的瓦片坐标，区分主地图与子场景
>
> 加载存档时，这些数据直接从快照 JSON 恢复到 session 内存，**不再查询共享的 SQLite 表**（如 `dialogues`），从而确保不同存档之间的对话历史、NPC 状态和全局标志完全隔离。
>
> `turn_index` 机制：每次玩家或 NPC 发言时，`dialogue_turn_counter` 自增并写入 `DialogueTurn.turn_index`。`_collect_full_dialogue_history()` 按 `turn_index` 排序，确保跨 NPC 的对话时间顺序正确，避免按 stage 或 NPC 遍历顺序错乱。
