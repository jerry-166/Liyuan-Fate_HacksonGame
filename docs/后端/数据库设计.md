# 《梨园生死》数据库设计

> **WAL 模式**：初始化时执行 `PRAGMA journal_mode=WAL;`，提升高频写入时的并发安全。
> **异步写入**：对话写入用 `asyncio.to_thread()` 包裹，避免阻塞事件循环。

---

## 建表语句

```sql
-- 启用 WAL 模式（应用启动时执行一次）
PRAGMA journal_mode=WAL;

-- sessions 表：会话元数据
CREATE TABLE sessions (
    session_id   TEXT PRIMARY KEY,
    player_name  TEXT NOT NULL DEFAULT '玩家',
    current_stage INTEGER NOT NULL DEFAULT 1,
    game_ended   INTEGER NOT NULL DEFAULT 0,  -- 0/1
    ending_type  TEXT,
    ending_data  TEXT,                         -- JSON: 缓存的结局评价
    deleted      INTEGER NOT NULL DEFAULT 0,   -- 0/1: 软删除
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- npc_states 表：NPC 状态快照
CREATE TABLE npc_states (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id   TEXT NOT NULL,
    npc_id       TEXT NOT NULL,
    relationship INTEGER NOT NULL DEFAULT 0,
    is_available INTEGER NOT NULL DEFAULT 1,
    current_greeting TEXT DEFAULT '',
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

-- dialogues 表：对话历史
CREATE TABLE dialogues (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id   TEXT NOT NULL,
    npc_id       TEXT NOT NULL,
    role         TEXT NOT NULL,  -- 'player' | 'npc'
    content      TEXT NOT NULL,
    options      TEXT,            -- JSON: NPC 回复的对话选项
    stage        INTEGER,
    summarized   INTEGER DEFAULT 0,
    summary      TEXT,
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_dialogue_session ON dialogues(session_id, npc_id);

-- events 表：已触发事件日志
CREATE TABLE events (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id   TEXT NOT NULL,
    event_id     TEXT NOT NULL,
    triggered_by TEXT,           -- npc_id or 'system'
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE INDEX idx_events_session ON events(session_id);

-- npc_memories 表：NPC 关键记忆（摘要预留）
CREATE TABLE npc_memories (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id   TEXT NOT NULL,
    npc_id       TEXT NOT NULL,
    content      TEXT NOT NULL,
    importance   INTEGER DEFAULT 1,
    created_at   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);
```

## 对话摘要预留字段

```sql
-- dialogues 表增加摘要相关字段
ALTER TABLE dialogues ADD COLUMN summarized BOOLEAN DEFAULT FALSE;
ALTER TABLE dialogues ADD COLUMN summary TEXT;
ALTER TABLE dialogues ADD COLUMN summary_at TIMESTAMP;
```
