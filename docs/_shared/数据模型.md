## GameSession

```python
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class GameSession:
    """一次完整的游戏会话（内存热数据）。"""
    session_id: str
    player_name: str
    api_key: Optional[str] = None     # 仅内存，不持久化
    model: Optional[str] = None       # 仅内存，None=使用 config 默认模型

    current_stage: int = 1
    npcs: dict = field(default_factory=dict)          # npc_id → NPCState
    events_triggered: set = field(default_factory=set)
    game_ended: bool = False
    ending_type: Optional[str] = None
    ending_data: Optional[dict] = None                # 缓存的结局评价

    # v2 章节驱动字段
    script_id: str = ""                 # 剧本 ID（如 liyuan_shengsi）
    current_chapter_id: Optional[str] = None
    chapter_defs: list = field(default_factory=list)  # 章节定义列表（只读）
    item_defs: list = field(default_factory=list)     # 物品定义列表（只读）
    persona_cache: dict = field(default_factory=dict) # NPC 人设缓存
    current_task: Optional[TaskInstance] = None
    inventory: list = field(default_factory=list)     # NarrativeItem 列表
    completed_chapters: list = field(default_factory=list)
    active_item: Optional[str] = None

    # 阶段判定辅助
    stage_llm_consecutive: int = 0   # LLM 连续判定推进的轮数（≥2 才切换）
    last_active_at: float = 0.0      # TTL 淘汰依据
    system_prompt: str = ""          # 动态构建的 system prompt
    dialogue_turn_counter: int = 0   # 对话全局序号（跨 NPC 时间排序）
    current_save_id: Optional[str] = None  # 当前存档 ID
    chapter_outlines: list = field(default_factory=list)  # AI 生成的章节大纲
```

## NPCState

```python
@dataclass
class NPCState:
    """单个 NPC 的运行时状态。"""
    id: str                    # "npc_chen"
    name: str                  # "陈师傅"
    role: str                  # "老琴师"
    scene: str                 # "tavern"
    position: dict             # {"x": int, "y": int}
    sprite_key: str            # "npc_chen_idle"
    relationship: int = 0      # -100 ~ 100
    is_available: bool = True
    current_greeting: str = ""
    dialogue_history: list = field(default_factory=list)  # 最近 N 轮（仅内存）
    last_options: list[str] = field(default_factory=list)
    dialogue_round_count: int = 0
    relationship_default: int = 0    # NPC 初始关系值

    def clamp_relationship(self) -> None:
        self.relationship = max(-100, min(100, self.relationship))

    def apply_delta(self, delta: int) -> None:
        self.relationship += delta
        self.clamp_relationship()
```

## SubTask（子任务）

```python
from enum import Enum

class SubTaskMode(str, Enum):
    DIALOGUE = "dialogue"              # 与 NPC 对话
    ACQUIRE_ITEM = "acquire_item"      # 获取物品
    SHOW_ITEM = "show_item"            # 向 NPC 展示物品
    DELIVER = "deliver"                # NPC A→主角→NPC B 传递链
    RELATION_THRESHOLD = "relation"    # 关系值达标
    EXPLORE = "explore"                # 到达指定场景

class SubTaskStatus(str, Enum):
    LOCKED = "locked"       # 前置未完成
    ACTIVE = "active"       # 可进行
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"

@dataclass
class SubTask:
    id: str
    title: str
    mode: SubTaskMode
    description: str = ""
    target_npc_id: Optional[str] = None
    deliver_to_npc_id: Optional[str] = None
    required_item_id: Optional[str] = None
    target_scene: Optional[str] = None
    relation_threshold: Optional[int] = None
    status: SubTaskStatus = SubTaskStatus.LOCKED
```

## TaskInstance（任务实例）

```python
@dataclass
class TaskInstance:
    task_id: str
    chapter_id: str
    chapter_name: str
    description: str
    sub_tasks: list[SubTask]
    related_npc_ids: list[str]
    generated_item_ids: list[str] = field(default_factory=list)
    npc_completion_votes: dict[str, bool] = field(default_factory=dict)
    is_completed: bool = False

    @property
    def completion_rate(self) -> float:
        """NPC 共识投票完成度 0.0~1.0，所有 NPC 投票后达到 1.0。"""
        total = len(self.related_npc_ids)
        if total == 0:
            return 1.0
        return sum(1 for v in self.npc_completion_votes.values() if v) / total
```

```python
    """叙事物品 — 承载剧情线索的可交互对象。"""

    is_key: bool = False
    location: Optional[dict] = None          # {"scene": ..., "position": {"col":..., "row":...}}
    is_discovered: bool = False

    related_npcs: list[str] = field(default_factory=list)
    npc_knowledge: dict[str, str] = field(default_factory=dict)  # NPC对此物品的特殊认知
    desc_source: str = "fixed"
    stage_relevance: list[int] = field(default_factory=list)
    source_npc: Optional[str] = None
    template_ref: Optional[str] = None
    holdable: bool = True
    acquire_method: str = ""
```
